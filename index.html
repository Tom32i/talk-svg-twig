Macros


<svg viewBox="0 0 20 20" xmlns="http://www.w3.org/2000/svg">
    {% for value in data %}
      {{ _self.bar(loop.index, value) }}
    {% endfor %}
</svg>

{% macro bar(x, height, width = 1, class = 'bar') %}
  <rect class="{{ class }}"
    x="{{ x - width / 2 }}" y="0"
    width="{{ width }}" height="{{ height }}" />
{% endmacro %}
